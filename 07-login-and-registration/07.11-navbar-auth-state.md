# Navbar with Auth State

Create components/Navbar.tsx. It reads auth state from Redux and shows either login/register links or the user's email and logout button.

```tsx
// components/Navbar.tsx
'use client';
import Link from 'next/link';
import { useAppSelector, useAppDispatch } from '@/store/store';
import { logout } from '@/store/authSlice';

export function Navbar() {
  const { user, isAuthenticated } = useAppSelector(s => s.auth);
  const dispatch = useAppDispatch();

  return (
    <nav style={{ display: 'flex', gap: '1rem', padding: '1rem', borderBottom: '1px solid #eee' }}>
      <Link href="/">EduFlow</Link>
      <Link href="/courses">Courses</Link>
      {isAuthenticated ? (
        <>
          <span>{user?.email}</span>
          <Link href="/dashboard">Dashboard</Link>
          <button onClick={() => dispatch(logout())}>Logout</button>
        </>
      ) : (
        <>
          <Link href="/login">Login</Link>
          <Link href="/register">Register</Link>
        </>
      )}
    </nav>
  );
}
```

Add Navbar to the root layout inside Providers:

```tsx
// app/layout.tsx
<Providers>
  <Navbar />
  {children}
</Providers>
```

✅ After login, the Navbar shows the email and Dashboard link.
✅ After logout (via Navbar button), the Navbar reverts to Login / Register.
