# Lesson Editor

Create components/LessonEditor.tsx. It shows the existing lessons and an add-lesson form.

```tsx
// components/LessonEditor.tsx — shape only
'use client';
import { useState } from 'react';
import api from '@/lib/api';
import { Lesson } from '@/types';

export function LessonEditor({ courseId, initialLessons }: { courseId: string; initialLessons: Lesson[] }) {
  const [lessons, setLessons] = useState(initialLessons);
  const [title, setTitle]     = useState('');
  const [duration, setDuration] = useState('0');

  async function addLesson() {
    const { data } = await api.post(`/api/instructor/courses/${courseId}/lessons`, {
      title,
      durationSeconds: Number(duration),
    });
    setLessons([...lessons, data]);
    setTitle('');
    setDuration('0');
  }

  async function deleteLesson(lessonId: string) {
    await api.delete(`/api/instructor/courses/${courseId}/lessons/${lessonId}`);
    setLessons(lessons.filter(l => l.id !== lessonId));
  }

  return (
    <section>
      <h2>Lessons ({lessons.length})</h2>
      {lessons.map((l, i) => (
        <div key={l.id} style={{ display: 'flex', justifyContent: 'space-between' }}>
          <span>{i + 1}. {l.title}</span>
          <button onClick={() => deleteLesson(l.id)}>Delete</button>
        </div>
      ))}
      <div>
        <input value={title}    onChange={e => setTitle(e.target.value)} placeholder="Lesson title" />
        <input value={duration} onChange={e => setDuration(e.target.value)} type="number" placeholder="Duration (seconds)" />
        <button onClick={addLesson}>Add Lesson</button>
      </div>
    </section>
  );
}
```
