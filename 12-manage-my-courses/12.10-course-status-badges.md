# Course Status Badges

Create a small StatusBadge component that shows the course status with a colour:

```tsx
// components/StatusBadge.tsx
const COLOURS: Record<string, string> = {
  draft:     '#6c757d',
  pending:   '#ffc107',
  published: '#28a745',
  rejected:  '#dc3545',
};

export function StatusBadge({ status }: { status: string }) {
  return (
    <span style={{
      background:   COLOURS[status] || '#6c757d',
      color:        '#fff',
      padding:      '0.25rem 0.5rem',
      borderRadius: 4,
      fontSize:     '0.75rem',
      fontWeight:   600,
    }}>
      {status.toUpperCase()}
    </span>
  );
}
```

Use StatusBadge in the instructor dashboard course list to show each course's current status at a glance.
