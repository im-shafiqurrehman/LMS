# Submit for Review Button

Add a Submit for Review button to the edit course page. It calls PATCH /api/instructor/courses/:id/submit.

```tsx
async function handleSubmit() {
  try {
    await api.patch(`/api/instructor/courses/${courseId}/submit`);
    setCourse(prev => prev ? { ...prev, status: 'pending' } : prev);
  } catch (err: any) {
    setError(err.response?.data?.error || 'Failed to submit');
  }
}

// In the UI, only show the button if status is 'draft' and there are lessons
{course?.status === 'draft' && lessons.length > 0 && (
  <button onClick={handleSubmit}>Submit for Review</button>
)}
```

After submitting, the edit form should become read-only. Show a "Pending review" banner instead of the edit controls:

```tsx
{course?.status === 'pending' && (
  <div style={{ background: '#fff3cd', padding: '0.75rem', borderRadius: 4 }}>
    This course is under review. Editing is disabled until the admin reviews it.
  </div>
)}
```
