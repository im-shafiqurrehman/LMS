# The User Schema

Open server/src/models/user.model.js and write the full User schema. Read each field decision before you type it.

```js
// src/models/user.model.js — shape only
import mongoose from 'mongoose';
import bcrypt   from 'bcrypt';  // npm install bcrypt

const userSchema = new mongoose.Schema(
  {
    email: {
      type:     String,
      required: [true, 'Email is required'],
      unique:   true,
      lowercase: true,
      trim:     true,
    },
    passwordHash: {
      type:     String,
      required: true,
      select:   false,   // never returned by queries by default
    },
    role: {
      type:    String,
      enum:    ['student', 'instructor', 'admin'],
      default: 'student',
    },
    emailVerifiedAt: { type: Date, default: null },
    verificationOtp: { type: String, select: false },
    otpExpiresAt:    { type: Date,   select: false },
  },
  { timestamps: true }
);

// Automatically hash passwordHash before every save that modifies it
userSchema.pre('save', async function (next) {
  if (!this.isModified('passwordHash')) return next();
  this.passwordHash = await bcrypt.hash(this.passwordHash, 12);
  next();
});

export const User = mongoose.model('User', userSchema);
```

**Key decisions:**
- select: false on passwordHash — it will never leak into API responses unless explicitly requested with .select('+passwordHash')
- emailVerifiedAt is a Date (null = unverified) — cleaner than a boolean because it records when verification happened
- The pre('save') hook owns all password hashing — route handlers never call bcrypt directly

Install bcrypt: npm install bcrypt && npm install --save-dev @types/bcrypt
