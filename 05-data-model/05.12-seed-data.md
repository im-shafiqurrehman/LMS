# Seed the Database

Create server/src/lib/seed.js. This script populates the database with enough data to develop against — an admin user, two categories, two instructor users, and two published courses with lessons.

```js
// src/lib/seed.js — shape only (you write the implementation)
import 'dotenv/config';
import mongoose from 'mongoose';
import { config } from '../config/index.js';
import { User }     from '../models/user.model.js';
import { Category } from '../models/category.model.js';
import { Course }   from '../models/course.model.js';
import { Lesson }   from '../models/lesson.model.js';

async function seed() {
  await mongoose.connect(config.databaseUrl);

  // Clear existing data
  await User.deleteMany({});
  await Category.deleteMany({});
  await Course.deleteMany({});
  await Lesson.deleteMany({});

  // Create admin
  const admin = await User.create({
    email: 'admin@eduflow.dev',
    passwordHash: 'Admin1234!',  // hashed by pre('save') hook
    role: 'admin',
    emailVerifiedAt: new Date(),
  });

  // Create categories
  const [webDev, dataScience] = await Category.insertMany([
    { name: 'Web Development', slug: 'web-development' },
    { name: 'Data Science',    slug: 'data-science' },
  ]);

  // Create instructor + course + lessons (implement the rest)
  console.log('Seed complete');
  await mongoose.disconnect();
}

seed().catch(console.error);
```

Add a seed script to package.json:
```json
"scripts": {
  "seed": "node src/lib/seed.js"
}
```

Run it: npm run seed

✅ After running, you should see documents in mongosh:

```
mongosh "mongodb://admin:yourpassword@localhost:27017/eduflow?authSource=admin"
use eduflow
db.users.find({}, { email: 1, role: 1 })
db.categories.find()
db.courses.find({}, { title: 1, status: 1 })
```
