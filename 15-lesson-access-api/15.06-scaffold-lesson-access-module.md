# Scaffold the Learn Module

```js
// src/modules/learn/learn.routes.js — shape only
import { Router }       from 'express';
import { learnService } from './learn.service.js';

export const learnRouter = Router();

learnRouter.get('/:courseId/lessons',            /* list lessons with video */ );
learnRouter.get('/:courseId/lessons/:lessonId',  /* single lesson with video */ );
learnRouter.post('/progress',                    /* mark lesson complete */ );
learnRouter.get('/:courseId/progress',           /* get course progress */ );
learnRouter.post('/certificates/:courseId',      /* issue certificate */ );
learnRouter.get('/certificates',                 /* list certificates */ );
learnRouter.get('/:courseId/questions',          /* list questions */ );
learnRouter.post('/:courseId/questions',         /* post question */ );
learnRouter.post('/questions/:qId/answers',      /* post answer (instructor) */ );
```
