# Test the Lesson Access API

```bash
BASE=http://localhost:3000

# 1. Login as enrolled student
STUDENT_TOKEN=$(...)  # login with enrolled student credentials

# 2. Get lesson list with video URLs
curl -s $BASE/api/learn/<courseId>/lessons \
  -H "Authorization: Bearer $STUDENT_TOKEN" | python3 -m json.tool
# Expected: lessons WITH videoUrl

# 3. Mark a lesson complete
curl -s -X POST $BASE/api/learn/progress \
  -H "Content-Type: application/json" \
  -H "Authorization: Bearer $STUDENT_TOKEN" \
  -d '{"courseId":"<id>","lessonId":"<id>"}'

# 4. Get progress
curl -s $BASE/api/learn/<courseId>/progress \
  -H "Authorization: Bearer $STUDENT_TOKEN" | python3 -m json.tool

# 5. Try to access lessons as non-enrolled student → 403
STUDENT2_TOKEN=$(...)  # login with a non-enrolled student
curl -s $BASE/api/learn/<courseId>/lessons \
  -H "Authorization: Bearer $STUDENT2_TOKEN"
# Expected: 403 Not enrolled
```
