# Category Filter

Create components/CategoryFilter.tsx. It fetches the category list from a new endpoint (add GET /api/categories to a new module) and renders filter buttons.

Add to the server (quick addition to the catalogue router):

```js
// In catalogue.routes.js
catalogueRouter.get('/categories', async (req, res, next) => {
  try {
    const categories = await Category.find().sort({ name: 1 });
    res.json(categories);
  } catch (err) { next(err); }
});
```

Frontend component:

```tsx
// components/CategoryFilter.tsx — shape only
'use client';
import { useEffect, useState } from 'react';
import api from '@/lib/api';

interface Props { selected: string; onSelect: (id: string) => void; }

export function CategoryFilter({ selected, onSelect }: Props) {
  const [categories, setCategories] = useState([]);

  useEffect(() => {
    api.get('/api/courses/categories').then(r => setCategories(r.data));
  }, []);

  return (
    <aside>
      <button onClick={() => onSelect('')} style={{ fontWeight: !selected ? 700 : 400 }}>All</button>
      {categories.map((c: any) => (
        <button key={c.id} onClick={() => onSelect(c.id)} style={{ fontWeight: selected === c.id ? 700 : 400 }}>
          {c.name}
        </button>
      ))}
    </aside>
  );
}
```

✅ Clicking a category re-fetches the courses for that category.
✅ Clicking "All" clears the filter.
