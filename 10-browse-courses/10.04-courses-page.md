# The Courses Page

Create app/courses/page.tsx. This is a client component (it uses state and effects):

```tsx
// app/courses/page.tsx — shape only
'use client';
import { useEffect, useState }        from 'react';
import { useAppDispatch, useAppSelector } from '@/store/store';
import { setCourses, appendCourses, setLoading } from '@/store/courseSlice';
import api from '@/lib/api';

export default function CoursesPage() {
  const dispatch = useAppDispatch();
  const { courses, loading, hasMore, nextCursor } = useAppSelector(s => s.courses);
  const [categoryId, setCategoryId] = useState('');
  const [q, setQ] = useState('');

  async function fetchCourses(opts: { reset?: boolean } = {}) {
    dispatch(setLoading(true));
    const params = new URLSearchParams({ limit: '12' });
    if (categoryId) params.set('categoryId', categoryId);
    if (q)          params.set('q', q);
    if (!opts.reset && nextCursor) params.set('cursor', nextCursor);

    const { data } = await api.get(`/api/courses?${params}`);
    if (opts.reset) {
      dispatch(setCourses(data));
    } else {
      dispatch(appendCourses(data));
    }
  }

  useEffect(() => { fetchCourses({ reset: true }); }, [categoryId, q]);

  return (
    <main>
      <h1>Courses</h1>
      {/* CategoryFilter, SearchInput, CourseGrid, LoadMore button */}
    </main>
  );
}
```

✅ On mount, the page fetches and displays courses from the API.
✅ Changing the category filter or search query re-fetches from the beginning.
