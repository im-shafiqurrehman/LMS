# Checkout Page

Create app/checkout/[courseId]/page.tsx. It creates a PaymentIntent on mount and renders the Stripe Elements form:

```tsx
// app/checkout/[courseId]/page.tsx — shape only
'use client';
import { loadStripe } from '@stripe/stripe-js';
import { Elements } from '@stripe/react-stripe-js';
import { CheckoutForm } from '@/components/CheckoutForm';
// ...
const stripePromise = loadStripe(process.env.NEXT_PUBLIC_STRIPE_PUBLISHABLE_KEY!);

export default function CheckoutPage() {
  // 1. Create payment intent on mount
  // 2. Pass clientSecret to Elements
  // 3. Render CheckoutForm inside Elements
  return (
    <Elements stripe={stripePromise} options={{ clientSecret }}>
      <CheckoutForm />
    </Elements>
  );
}
```
