# Stripe Elements Form

Create components/CheckoutForm.tsx using Stripe's pre-built PaymentElement:

```tsx
// components/CheckoutForm.tsx — shape only
'use client';
import { PaymentElement, useStripe, useElements } from '@stripe/react-stripe-js';

export function CheckoutForm({ courseId }: { courseId: string }) {
  const stripe   = useStripe();
  const elements = useElements();

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    if (!stripe || !elements) return;

    const { error } = await stripe.confirmPayment({
      elements,
      confirmParams: {
        return_url: `${window.location.origin}/checkout/success?courseId=${courseId}`,
      },
    });

    if (error) setError(error.message ?? 'Payment failed');
  }

  return (
    <form onSubmit={handleSubmit}>
      <PaymentElement />
      <button type='submit'>Pay Now</button>
    </form>
  );
}
```

Stripe's PaymentElement handles card input, 3DS, and all payment method UI automatically.
