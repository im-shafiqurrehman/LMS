# Enrollment Guard

Create components/EnrollmentGuard.tsx. The lesson player (chapter 16) uses it to redirect unenrolled users away from lesson pages.

```tsx
'use client';
import { useEffect, useState } from 'react';
import { useRouter }           from 'next/navigation';
import api from '@/lib/api';
import { useAppSelector } from '@/store/store';

export function EnrollmentGuard({ courseId, children }: { courseId: string; children: React.ReactNode }) {
  const { isAuthenticated } = useAppSelector(s => s.auth);
  const router     = useRouter();
  const [checked, setChecked] = useState(false);
  const [enrolled, setEnrolled] = useState(false);

  useEffect(() => {
    if (!isAuthenticated) { router.replace('/login'); return; }
    api.get(`/api/enrollments/check/${courseId}`)
      .then(r => { setEnrolled(r.data.enrolled); setChecked(true); })
      .catch(() => router.replace('/courses'));
  }, [courseId, isAuthenticated]);

  if (!checked) return <p>Verifying access...</p>;
  if (!enrolled) { router.replace(`/courses/${courseId}`); return null; }
  return <>{children}</>;
}
```
