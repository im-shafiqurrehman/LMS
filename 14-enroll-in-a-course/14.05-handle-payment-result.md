# Handle Payment Result

After the payment completes, Stripe redirects to the return_url. Create app/checkout/success/page.tsx:

```tsx
// app/checkout/success/page.tsx — shape only
'use client';
import { useEffect, useState } from 'react';
import { useSearchParams, useRouter } from 'next/navigation';
import api from '@/lib/api';

export default function CheckoutSuccessPage() {
  const params   = useSearchParams();
  const courseId = params.get('courseId');
  const router   = useRouter();
  const [status, setStatus] = useState('Confirming your enrollment...');

  useEffect(() => {
    let attempts = 0;
    const poll = setInterval(async () => {
      const { data } = await api.get(`/api/enrollments/check/${courseId}`);
      if (data.enrolled) {
        clearInterval(poll);
        setStatus('Enrolled! Redirecting...');
        setTimeout(() => router.push('/dashboard/my-learning'), 1500);
      }
      if (++attempts > 10) {
        clearInterval(poll);
        setStatus('Enrollment pending — check My Learning in a moment.');
      }
    }, 1000);
    return () => clearInterval(poll);
  }, [courseId]);

  return <main><h1>{status}</h1></main>;
}
```

Polling is simpler than websockets for this case. 10 attempts × 1 second = 10 second max wait. Stripe webhook delivery is typically under 2 seconds.
