# My Learning Page

Create app/dashboard/my-learning/page.tsx:

```tsx
'use client';
import { useEffect, useState } from 'react';
import Link from 'next/link';
import api from '@/lib/api';
import { AuthGuard } from '@/components/AuthGuard';

export default function MyLearningPage() {
  const [enrollments, setEnrollments] = useState([]);

  useEffect(() => {
    api.get('/api/enrollments/mine').then(r => setEnrollments(r.data));
  }, []);

  return (
    <AuthGuard>
      <main>
        <h1>My Learning</h1>
        {enrollments.length === 0 && <p>No enrollments yet. <Link href='/courses'>Browse courses</Link></p>}
        {enrollments.map((e: any) => (
          <div key={e.id} style={{ border: '1px solid #eee', padding: '1rem', marginBottom: '0.5rem' }}>
            <h2>{e.courseId.title}</h2>
            <p>Paid: ${e.amountPaid}</p>
            <Link href={`/learn/${e.courseId.id}`}><button>Continue Learning</button></Link>
          </div>
        ))}
      </main>
    </AuthGuard>
  );
}
```
