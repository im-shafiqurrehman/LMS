# Mark Lesson Complete (Refined)

Update the markLessonComplete service to return richer data for the frontend:

```js
async function markLessonComplete(studentId, courseId, lessonId) {
  const enrollment = await requireEnrollment(studentId, courseId);

  await ProgressRecord.findOneAndUpdate(
    { enrollmentId: enrollment._id, lessonId },
    { completedAt: new Date() },
    { upsert: true }
  );

  const [totalLessons, completedCount] = await Promise.all([
    Lesson.countDocuments({ courseId }),
    ProgressRecord.countDocuments({ enrollmentId: enrollment._id }),
  ]);

  const percentage     = Math.round((completedCount / totalLessons) * 100);
  const courseComplete = completedCount >= totalLessons;

  if (courseComplete) {
    // Auto-issue certificate
    await Certificate.findOneAndUpdate(
      { enrollmentId: enrollment._id },
      { issuedAt: new Date() },
      { upsert: true }
    );
  }

  return { completedCount, totalLessons, percentage, courseCompleted: courseComplete };
}
```
