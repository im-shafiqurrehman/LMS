# Generate PDF Certificate (Bar-Raiser)

Install pdfkit: npm install pdfkit

```js
import PDFDocument from 'pdfkit';

export async function generateCertificatePDF({ studentName, courseTitle, issuedAt }) {
  return new Promise((resolve, reject) => {
    const doc    = new PDFDocument({ size: 'A4', layout: 'landscape' });
    const chunks = [];

    doc.on('data',  chunk => chunks.push(chunk));
    doc.on('end',   () => resolve(Buffer.concat(chunks)));
    doc.on('error', reject);

    doc.fontSize(40).text('Certificate of Completion', { align: 'center' });
    doc.moveDown();
    doc.fontSize(24).text(`Awarded to: ${studentName}`, { align: 'center' });
    doc.fontSize(20).text(`Course: ${courseTitle}`, { align: 'center' });
    doc.fontSize(16).text(`Issued: ${new Date(issuedAt).toLocaleDateString()}`, { align: 'center' });
    doc.end();
  });
}
```

After generating the Buffer, upload to Cloudinary (or return as a base64 string for a quick demo) and store the URL in certificate.pdfUrl.
