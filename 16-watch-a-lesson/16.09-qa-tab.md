# Q&A Tab

Add a Q&A section below the video player:

```tsx
function QATab({ courseId }: { courseId: string }) {
  const [questions, setQuestions] = useState([]);
  const [content, setContent]     = useState('');

  useEffect(() => {
    api.get(`/api/learn/${courseId}/questions`).then(r => setQuestions(r.data));
  }, [courseId]);

  async function postQuestion() {
    const { data } = await api.post(`/api/learn/${courseId}/questions`, { content });
    setQuestions(prev => [data, ...prev]);
    setContent('');
  }

  return (
    <section>
      <h2>Q&A</h2>
      <textarea value={content} onChange={e => setContent(e.target.value)} placeholder="Ask a question..." />
      <button onClick={postQuestion}>Post Question</button>
      {questions.map((q: any) => <div key={q.id}>{q.userId.email}: {q.content}</div>)}
    </section>
  );
}
```
