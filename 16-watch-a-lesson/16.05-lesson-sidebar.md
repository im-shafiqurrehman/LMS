# Lesson Sidebar

Create components/LessonSidebar.tsx. It shows all course lessons with a checkmark on completed ones:

```tsx
export function LessonSidebar({ courseId, lessons, completedIds, currentLessonId }: {...}) {
  return (
    <aside style={{ width: 280, borderRight: '1px solid #eee', overflowY: 'auto' }}>
      <h2>Course Content</h2>
      {lessons.map((lesson, i) => (
        <Link key={lesson.id} href={`/learn/${courseId}/${lesson.id}`}>
          <div style={{
            padding: '0.75rem',
            background: currentLessonId === lesson.id ? '#f0f0f0' : 'transparent',
            display: 'flex', justifyContent: 'space-between',
          }}>
            <span>{i + 1}. {lesson.title}</span>
            {completedIds.includes(lesson.id) && <span style={{ color: '#28a745' }}>✓</span>}
          </div>
        </Link>
      ))}
    </aside>
  );
}```
